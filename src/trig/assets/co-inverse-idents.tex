\begin{equation}
\begin{aligned}
    \sin^{-1}\frac{1}{x} &= \csc^{-1}x & \text{if}\; & -1 \leq x \leq 1 \\
    &\\
    \cos^{-1}\frac{1}{x} &= \sec^{-1}x & \text{if}\; & -1 \leq x \leq 1 \\
    &\\
    \frac{\pi}{2} - \tan^{-1}x &= \cot^{-1}x & \text{if}\; & x \in \mathbf{R} \\
\end{aligned}
\end{equation}
